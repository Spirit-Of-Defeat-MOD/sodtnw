FRA_support_italian_socialists = {
	enable = {
		tag = FRA
		country_exists = ITS
		ITS = { has_war_with = ITM }
		ITS = { has_war_with = ITA }
	}
	abort = {
		OR = {
			NOT = { country_exists = ITS }
		}
	}
	
	ai_strategy = {
		type = support
		id = "ITS"
		value = 1000
	}
	ai_strategy = {
		type = send_volunteers_desire

	}
}
