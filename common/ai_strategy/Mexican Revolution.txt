###Experimental (USA will support whoever controls mexico city except for socialists )

USA_support_feds = { ##if they own the city
	enable = {
		tag = USA
		country_exists = MEX
		MEX = { owns_state = 277 }
	}
	abort = {
		OR = {
			NOT = { country_exists = MEX }
		}
	}
	
	ai_strategy = {
		type = support
		id = "MEX"
		value = 1000
	}
	ai_strategy = {
		type = send_volunteers_desire
		id = "MEX"
		value = 200
	}	
	ai_strategy = {
		type = send_lend_lease_desire
		id = "MEX"
		value = 200
	}
}
