on_actions = {
	on_startup = {
		effect = {
			GER = { country_event = germany.1 }
			SOV = { country_event = soviet.1 }
			ITA = { country_event = italy.1 }
			ENG = { country_event = uk.2 }
			FRA = { country_event = france.1 }
			SPR = { country_event = spain.1 }
			POL = { country_event = poland.1 }
			RHI = { country_event = rhine_political.0 }
			TUR = { country_event = turkey.1 }
			CRO = { country_event = croatia.1 }
			SER = { country_event = serbia.1 }
			GRE = { country_event = greece.1 }
			BUL = { country_event = bulgaria.1 }
			IRE = { country_event = ireland.1 }
			FIC = { country_event = indochina.1 }
			#ALB = { country_event = albania.1 }
			#AST = { country_event = ? }
			#SIA = { country_event = siam.1 }
			
			every_country = {
				if = {
					limit = {
						exists = yes
						is_ai = no
					}
					country_event = { id = intro.1 }
				}
			}
			
			MEX = { set_cosmetic_tag = MEX_CIVIL_WAR }
			set_global_flag = mexican_revolution
		}
	}
	
	on_startup = { #Mexico Cores, this is so factions can gain manpower once they annex someone
		effect = {
			every_state = {
				limit = {
					is_core_of = MEX
				}
				add_core_of = CED
				add_core_of = MCT
				add_core_of = PNR
				add_core_of = PLM
				add_core_of = VIL
			}
			
			every_state = {
				limit = {
					is_claimed_by = MEX
				}
				add_claim_by = CED
				add_claim_by = MCT
				add_claim_by = PNR
				add_claim_by = PLM
				add_claim_by = VIL
			}
		}
	}
	
	on_annex = { #Should remove cores of dead factions so no revivals mid civil war
		effect = {
			every_state = {
				limit = {
					is_core_of = FROM
				}
				remove_core_of = FROM
			}
			
			every_state = {
				limit = {
					is_claimed_by = FROM
				}
				remove_claim_by = FROM
			}
		}
	}
	
	
	on_army_leader_won_combat = { 
		effect = {
			if = {
				limit = {
					FROM = {
						AND = {
							original_tag = SER
							has_government = communism
							has_civil_war = yes
						}
					}
					NOT = { 
						has_global_flag = SER_battle_of_prokuplje
					}
				}
				set_global_flag = SER_battle_of_prokuplje
				news_event = {
					id = serbianews.2
					hours = 2
					random = 2
				}
			}
			if = {
				limit = {
					FROM = {
						AND = {
							original_tag = SER
							has_government = utilitarian_despot
							has_civil_war = yes
						}
					}
					NOT = { 
						has_global_flag = SER_battle_of_prokuplje
					}
				}
				set_global_flag = SER_battle_of_prokuplje
				news_event = {
					id = serbianews.3
					hours = 2
					random = 2
				}
			}
		} 
	}
	on_civil_war_end = { 
		effect = {
			if = {
				limit = {
					ROOT = {
						original_tag = SER
						has_government = communism
					}
				}
				set_global_flag = serbcw_end
				set_global_flag = communistsercw_win
				news_event = { id = serbianews.4 random = 6 }
			}
			if = {
				limit = {
					ROOT = {
						original_tag = SER
						has_government = neutrality
					}
				}
				set_global_flag = serbcw_end
				set_global_flag = chetniksercw_win
				news_event = { id = serbianews.5 random = 6 }
			}
		}
	}
	on_capitulation = { 
		effect = {
			if = {
				limit = {
					ROOT = {
						original_tag = CRO
						has_war_with = SER
					}
				}
				SER = {
					country_event = sod_peace.2
				}
			}
			if = {
				limit = {
					ROOT = {
						original_tag = SER
						has_war_with = CRO
					}
				}
				CRO = {
					country_event = sod_peace.1
				}
			}
		} 
	}
}
