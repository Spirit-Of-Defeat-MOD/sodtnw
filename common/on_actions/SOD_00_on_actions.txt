on_actions = {
	on_startup = {
		effect = {
			GER = {
				country_event = germany.1
			}
			ITA = {
				country_event = italy.1
			}
			CRO = {
				country_event = croatia.2
			}
			SER = {
				country_event = serbia.1
			}
			every_country = {
				if = {
					limit = {
						exists = yes
						is_ai = no
					}
					country_event = { id = intro.1 }
				}
			}
		}
	}
    on_daily = {
        effect = {
            if = {
                limit = {
                    any_country = {
                        exists = no
                        has_country_flag = statistics_1
                    }
                }
                random_country = {
                    limit = {
                        exists = yes
                    }
                }
            }
        }
    }
	on_army_leader_won_combat = { 
		effect = {
			if = {
				limit = {
					FROM = {
						AND = {
							original_tag = SER
							has_government = communism
						}
						has_civil_war = yes
					}
					NOT = { 
						has_global_flag = SER_battle_of_prokuplje
					}
				}
				set_global_flag = SER_battle_of_prokuplje
				news_event = {
					id = serbianews.2
					hours = 2
					random = 2
				}
			}
			if = {
				limit = {
					FROM = {
						AND = {
							original_tag = SER
							has_government = utilitarian_despot
						}
						has_civil_war = yes
					}
					NOT = { 
						has_global_flag = SER_battle_of_prokuplje
					}
				}
				set_global_flag = SER_battle_of_prokuplje
				news_event = {
					id = serbianews.3
					hours = 2
					random = 2
				}
			}
		} 
	}
	on_civil_war_end = { 
		effect = {
			if = {
				limit = {
					ROOT = {
						original_tag = SER
						has_government = communism
					}
				}
				set_global_flag = serbcw_end
				set_global_flag = communistsercw_win
				news_event = { id = serbianews.4 random = 6 }
			}
			if = {
				limit = {
					ROOT = {
						original_tag = SER
						has_government = neutrality
					}
				}
				set_global_flag = serbcw_end
				set_global_flag = chetniksercw_win
				news_event = { id = serbianews.5 random = 6 }
			}
		}
	}
}
