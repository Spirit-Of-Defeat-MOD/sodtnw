scripted_gui = {
	Language_Topbar_Window = {
		context_type = player_context
		window_name = "Language_Topbar_Window"
		parent_window_token = top_bar

		visible = {
			always = yes
		}

		effects = {
			Language_Window_Enabler_click = {
			if = {
				limit = {	
					NOT = {
						has_country_flag = Language_Window_Open
					}
				}
				set_country_flag  = Language_Window_Open
				}
			}
		}
	}
	
	Language_Window = {
		context_type = player_context
		window_name = "Language_Window"
		
		visible = {
			has_country_flag = Language_Window_Open
		}
		
		effects = {
			close_button_click = {
				clr_country_flag  = Language_Window_Open
			}
			
			wade_giles_romanization_click = {
				set_global_flag = wade_giles_romanization_set
				clr_global_flag = pinyin_romanization_set
				rename_chinese_states = yes
			}
			
			pinyin_romanization_click = {
				set_global_flag = pinyin_romanization_set
				clr_global_flag = wade_giles_romanization_set
				rename_chinese_states_pinyin
			}
		}
		triggers = {
		
			wade_giles_romanization_click_enabled = {
				NOT = { has_global_flag = wade_giles_romanization_set }
			}
			pinyin_romanization_click_enabled = {
				NOT = { has_global_flag = pinyin_romanization_set }
			}
		}
	}
}
