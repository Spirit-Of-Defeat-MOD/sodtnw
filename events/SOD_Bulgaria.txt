# Help Croatia
country_event = {
	id = bulgaria.2
    title = bulgaria.2.t
	desc = bulgaria.2.d
	picture = GFX_report_event_handschar_division
	
	fire_only_once = yes
	
	is_triggered_only = yes
	
	option = {
		name = bulgaria.2.a
		CRO = {
			create_faction = "Balkan League"
			add_to_faction = BUL
			if = {
				limit = {
					MAC = {
						is_puppet_of = BUL
					}
				}
				add_to_faction = MAC
			}
		}
	}
	
	option = {
		name = bulgaria.2.b
		CRO = {
			add_opinion_modifier = {
				target = BUL
				modifier = CRO_BUL_refuse_alliance
			}
		}
	}
}
