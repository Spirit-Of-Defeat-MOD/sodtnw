##Puebla Militia
country_event = {
	id = mexmilitias.1
	title = mexmilitias.1.t
	desc = mexmilitias.generic
	picture = GFX_event_Generic_mexmilitias

	is_triggered_only = yes
	fire_only_once = yes
	
	option = {
		name = mexmilitias.raise
		set_global_flag = puebla_militia_raised
		
		#Units
		division_template = {
			name = "Milicia"
			division_names_group = SPAN_INF_01
			is_locked = yes
			regiments = {
				infantry = { x = 0 y = 0 }
				infantry = { x = 0 y = 1 }
				infantry = { x = 1 y = 0 }
				infantry = { x = 1 y = 1 }
			}
		}
		
		800 = {
			create_unit = {
				division = "name = \"1. Milicia de Puebla\" division_template = \"Milicia\"  start_experience_factor = 0.1"
				allow_spawning_on_enemy_provs = yes
				owner = ROOT
				prioritize_location = 8089
			}
			
			create_unit = {
				division = "name = \"2. Milicia de Puebla\" division_template = \"Milicia\"  start_experience_factor = 0.1"
				allow_spawning_on_enemy_provs = yes
				owner = ROOT
				prioritize_location = 8089
			}
		}
	}
}
