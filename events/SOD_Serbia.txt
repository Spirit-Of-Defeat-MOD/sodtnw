		days = 5
	}

		name = "Already?!"

			if = {
				limit = {
					MNT = {

					}

				target = MNT

			}

				add_manpower = 30000
				army_experience = 50
				add_political_power = 150
				load_oob = "MNT_1936"
				set_war_support = 1
				add_command_power = 75
			}
		}
	}
}
