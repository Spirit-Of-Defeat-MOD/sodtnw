# Fall of Sarajevo
news_event = {
	id = serbianews.4
	title = serbianews.4.t
	desc = serbianews.4.d
	picture = GFX_news_event_hungarian_soldiers
	
	major = yes
	
	fire_only_once = yes
	
	trigger = {
		104 = { is_controlled_by = SER }
		104 = { is_owned_by = CRO }
		SER = { has_war_with = CRO }
		NOT = { has_global_flag = fall_of_sarajevo }
	}
	
	immediate = {
		set_global_flag = fall_of_sarajevo
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = "Onwards!"
		trigger = {
			original_tag = SER
		}
	}
	option = {
		name = "This is a disaster!"
		trigger = {
			original_tag = CRO
		}
	}
	option = {
		name = "This is getting out of hand."
		trigger = {
			NOT = {
				original_tag = CRO
				original_tag = SER
			}
		}
	}
}
