add_namespace = weapon_purchasing
country_event = {
	id = weapon_purchasing.1
	title = weapon_purchasing.1.t
	desc = weapon_purchasing.1.d
	picture = news_event_german_soldiers

	is_triggered_only = yes
	
	option = { #UK
		name = weapon_purchasing.1.a
		add_political_power = -150
		add_equipment_to_stockpile = {
			type = infantry_equipment_1
			amount = 1500
			producer = ENG
		}
		ENG = { add_offsite_building = { type = arms_factory level = 1 } }
		add_opinion_modifier = { target = ENG modifier = bought_our_weapons }
    
    }
	
	option = { #USA
